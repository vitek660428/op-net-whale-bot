// api/bot.js
const TelegramBot = require('node-telegram-bot-api');

const token = process.env.TELEGRAM_TOKEN; // токен из переменных окружения
const bot = new TelegramBot(token);

module.exports = async (req, res) => {
  try {
    if (req.method === 'POST') {
      const update = req.body;

      if (!update || !update.message) return res.status(200).send('No message');

      const chatId = update.message.chat.id;
      const text = update.message.text;

      if (text === '/start') {
        await bot.sendMessage(chatId, '🚀 Бот запущен!');
      } else if (text === '/ping') {
        await bot.sendMessage(chatId, '🏓 pong');
      }

      return res.status(200).send('OK');
    } else {
      return res.status(200).send('Telegram bot webhook endpoint');
    }
  } catch (err) {
    console.error(err);
    return res.status(500).send('Error');
  }
};
